% Define disease inheritance probabilities
% 1 = dominant, 2 = recessive
disease_inheritance(marker1, 1).
disease_inheritance(marker1, 2).

% Predicate for disease transmission
transmits_disease(Person, Disease) :-
    has_marker(Person, Marker),
    disease_inheritance(Marker, Inheritance),
    (
        Inheritance = 1, % If the disease is dominant
        (
            has_marker(Person, Disease) ; % Person has the disease marker
            (parent(Parent, Person), transmits_disease(Parent, Disease)) % Or parent transmits the disease
        )
    ;
        Inheritance = 2, % If the disease is recessive
        (
            has_marker(Person, Disease) ; % Person has the disease marker
            (
                parent(Parent1, Person),
                parent(Parent2, Person),
                transmits_disease(Parent1, Disease),
                transmits_disease(Parent2, Disease)
            ) % Both parents transmit the disease
        )
    ).
